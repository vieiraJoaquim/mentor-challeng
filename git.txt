Configurar Git

git config --global user.name "Vieira Joaquim"
git config --global user.email "vjoaquim1@gmail.com"
git config --global core.editor "vs code"

git config user.name // para saber o nome do usuário logado.
git config --list // para ver tudo q foi configurado.

git init // para iniciar o git na pasta desejada.
git status // para saber o estado atual do arquivo em que estamos trabalhando.
git add -A // "salva" todas as mudanças feitas no arquivo.
git add [nome do arquivo] // para add um arquivo específico.
git commit -m "Mensagem aqui" // salva no repositório local todas as mudanças.
git commit -am "Mensagem aqui" // para comitar sem precisar fazer o gir add.
git log // a listagem de todos os commits.

git branch // para listar todos os brenchs.
git branch [nome da brench] // para criar um novo brench.
git checkout [nome do brench] // para mudar o brench.

git diff // para saber exatamente o que foi alterado.
git diff --name-only // para saber exatamente todos os arquivos que foram modificados.
git diff [nome do arquivo] // saber exatamente o arquivo que foi alterado.
